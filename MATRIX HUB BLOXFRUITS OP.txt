-- Funções principais
local function AutoFarm()
    while getgenv().Settings.AutoFarm do
        -- Lógica de farm (similar às versões anteriores)
        task.wait()
    end
end

local function AutoMastery()
    while getgenv().Settings.AutoMastery do
        local key = getgenv().Settings.AutoMasteryType == "Melee" and "Z" 
                  or getgenv().Settings.AutoMasteryType == "Sword" and "X" 
                  or "C"
        game:GetService("VirtualInput"):SendKeyEvent(true, key, false, game)
        task.wait(0.2)
    end
end

local function Fly()
    local plr = game.Players.LocalPlayer
    while getgenv().Settings.Fly do
        if plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
            local root = plr.Character.HumanoidRootPart
            root.Velocity = Vector3.new(0, 0, 0)
            root.CFrame = root.CFrame + (root.CFrame.LookVector * getgenv().Settings.FlySpeed/100)
        end
        task.wait()
    end
end
